
model UserIndex {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  symbol    String   // e.g., "XBANK.IS"
  name      String   // e.g., "BIST Bankacılık"
  order     Int      @default(0)
  createdAt DateTime @default(now())
  
  @@unique([userId, symbol])
  @@index([userId, order])
}
